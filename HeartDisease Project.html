# Compute group means and SE
group_summary = hd_data %>%
  group_by(hd_binary) %>%
  summarise(
    mean_thalach = mean(thalach, na.rm = TRUE),
    sd_thalach = sd(thalach, na.rm = TRUE),
    n = n(),
    se_thalach = sd_thalach / sqrt(n)
  )
